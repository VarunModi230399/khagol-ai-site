--- ./components/3D/KhagolPlanet.tsx ---
import { useFrame } from '@react-three/fiber'
import { Ring, useTexture } from '@react-three/drei'
import { useRef } from 'react'
import type { Mesh } from 'three'
import { PLANETS } from '../../config/planets'
import { OrbitingPlanet } from './OrbitingPlanet'

export function KhagolPlanet() {
  const meshRef = useRef<Mesh | null>(null)
  const logoMap = useTexture('/logo.png')

  useFrame((state, delta) => {
    if (meshRef.current) meshRef.current.rotation.y += delta * 0.3
  })

  return (
    <group>
      <mesh ref={meshRef}>
        <sphereGeometry args={[1.2, 128, 128]} />
        <meshStandardMaterial
          color="#1a237e"
          metalness={0.2}
          roughness={0.3}
          emissive="#00bcd4"
          emissiveIntensity={0.2}
          map={logoMap}
        />
      </mesh>
      <mesh scale={[1.12, 1.12, 1.12]}>
        <sphereGeometry args={[1, 64, 64]} />
        <meshBasicMaterial color="#00bcd4" transparent opacity={0.3} />
      </mesh>

      <Ring args={[2.2, 2.25, 64]} rotation={[Math.PI / 2, 0, 0]}>
        <meshStandardMaterial
          color="#00bcd4"
          emissive="#00bcd4"
          emissiveIntensity={0.5}
          transparent
          opacity={0.8}
        />
      </Ring>

      {PLANETS.map((planet) => (
        <group key={planet.id} rotation={[planet.tiltX, planet.tiltY, 0]}>
          <Ring
            args={[planet.radius, planet.radius + 0.02, 64]}
            rotation={[Math.PI / 2, 0, 0]}
          >
            <meshBasicMaterial color={planet.color} transparent opacity={0.6} />
          </Ring>
          <OrbitingPlanet planet={planet} />
        </group>
      ))}
    </group>
  )
}

--- ./components/3D/OrbitingPlanet.tsx ---
import { useFrame } from '@react-three/fiber'
import { Text } from '@react-three/drei'
import { useRef, useState } from 'react'
import type { Mesh } from 'three'
import type { PlanetConfig } from '../../types/planet'

interface OrbitingPlanetProps {
  planet: PlanetConfig
}

export function OrbitingPlanet({ planet }: OrbitingPlanetProps) {
  const meshRef = useRef<Mesh | null>(null)
  const [hovered, setHovered] = useState(false)

  useFrame((state, delta) => {
    if (!meshRef.current) return
    const time = state.clock.elapsedTime
    meshRef.current.position.x =
      Math.cos(time * planet.speed + planet.angle) * planet.radius
    meshRef.current.position.z =
      Math.sin(time * planet.speed + planet.angle) * planet.radius
    meshRef.current.rotation.y += delta * 0.5
  })

  return (
    <group
      ref={meshRef}
      onPointerEnter={() => setHovered(true)}
      onPointerLeave={() => setHovered(false)}
    >
      <mesh scale={hovered ? 1.3 : 1}>
        <sphereGeometry args={[0.3, 64, 64]} />
        <meshStandardMaterial
          color={planet.color}
          metalness={0.1}
          roughness={0.4}
          emissive={planet.color}
          emissiveIntensity={hovered ? 0.4 : 0.15}
        />
      </mesh>
      <mesh scale={hovered ? [1.25, 1.25, 1.25] : [1.1, 1.1, 1.1]}>
        <sphereGeometry args={[1, 32, 32]} />
        <meshBasicMaterial color={planet.color} transparent opacity={0.2} />
      </mesh>
      <Text
        position={[0, 0.7, 0]}
        fontSize={0.12}
        color={hovered ? '#00bcd4' : '#ffffff66'}
        anchorX="center"
        anchorY="bottom"
      >
        {planet.name}
      </Text>
    </group>
  )
}
